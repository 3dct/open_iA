
SET (BUNDLE_APP "\${CMAKE_INSTALL_PREFIX}/open_iA${CMAKE_EXECUTABLE_SUFFIX}")

	SET (BUNDLE_LIBS \"${BUNDLE_LIBS}\")
	file(GLOB_RECURSE QTPLUGINS

	include(BundleUtilities)
	fixup_bundle(\"${BUNDLE_APP}\" \"\${BUNDLE_LIBS}\" \"${BUNDLE_DIRS}\")
	")
